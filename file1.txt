// PROGRAM: Temperature Converter
#include <stdio.h>
#include <conio.h>

int main() {
    float f, c;

    printf("Farenheit: "); scanf("%f", &f);

    c = (f - 32) * 5 / 9;

    printf("%.2fF = %.2fC\n", f, c);

    getch();
    return 0;
}
